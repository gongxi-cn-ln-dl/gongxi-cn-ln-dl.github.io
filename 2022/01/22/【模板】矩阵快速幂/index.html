矩阵快速幂

  • 顾名思义，其实就是矩阵乘法+快速幂

  • typedef long long ll;
const int maxN = 110;
const int MOD = (1e9+7);
int n;

struct Matrix{
    ll d[maxN][maxN];
    Matrix(){memset(d,0,sizeof(d));}
    void init(){
        for(int i = 1;i<=n;i++){
            d[i][i] = 1;
        }
    }
    friend Matrix operator * (Matrix A,Matrix B){
        Matrix C;
        for(int i = 1;i<=n;i++){
            for(int j = 1;j<=n;j++){
                for(int k = 1;k<=n;k++){
                    C.d[i][k] += (A.d[i][j] * B.d[j][k]) % MOD;
                    C.d[i][k] %= MOD;
                }
            }
        }
        return C;
    }
};

Matrix quickPow(Matrix A,ll b){
    Matrix B;
    B.init();
    while(b){
        if(b&1) B = B * A;
        A = A * A;
        b /= 2;
    }
    return B;
}
